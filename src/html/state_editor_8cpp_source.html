341 void StateEditor::RecomputeCollisionRectangles() {
342  TileMap &tm = level.tileMap;
343  vector<int> &coll=level.collisionLayer;
344  int mapWidth=tm.GetWidth();
345  int mapHeight=tm.GetHeight();
346 
347  grouped.clear();
348  grouped.resize(mapWidth*mapHeight);
349  FOR(y,mapHeight) {
350  FOR(x,mapWidth) {
351  int ind = x+(y*mapWidth);
352  grouped[ind]=make_pair(ii{x,y},ii{x,y});
353  }
354  }
355 
356  auto join = [this,mapWidth,mapHeight](int ind1,int ind2) {
357  ii beg=min(grouped[ind1].first, grouped[ind2].first);
358  ii end=max(grouped[ind1].second,grouped[ind2].second);
359 
360  int ind3=beg.first+(beg.second*mapWidth);
361  int ind4=end.first+(end.second*mapWidth);
362 
363  grouped[ind1]=grouped[ind2]=grouped[ind3]=grouped[ind4]=make_pair(beg,end);
364  };
365 
366  FOR(y,mapHeight) {
367  FOR(x,mapWidth) {
368  int ind = x+(y*mapWidth);
369  if (coll[ind]==EMPTY_TILE)grouped[ind]=make_pair(ii{0,0},ii{0,0});
370  else if (x+1 < mapWidth && coll[ind] == coll[1+ind]) {
371  join(ind,ind+1);
372  }
373  }
374  }
375 
376  FOR(y,mapHeight-1) {
377  FOR(x,mapWidth) {
378  int ind1 = x+( y *mapWidth);
379  int ind2 = x+((y+1)*mapWidth);
380  if (grouped[ind1].first.first == grouped[ind2].first.first &&
381  grouped[ind1].second.first == grouped[ind2].second.first) {
382  join(ind1,ind2);
383  }
384  }
385  }
