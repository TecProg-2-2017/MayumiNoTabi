713  Vec2 move;
714 
715  if (totMove==Vec2{})return;
716 
717  move.x = a.Collides(b,{totMove.x,0.0f},move).x;
718  if (move.x != totMove.x)speed.x=0.0f;

723  if (totMove.y>0)mem->ints["onAir"]=0;

725  else if (equals(totMove.y,0.0f) && equals(a.Collides(b,{0.0f,10},move).y,0.0f))mem->ints["onAir"]=0;
